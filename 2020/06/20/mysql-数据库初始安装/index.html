mysql 数据库初始安装

Posted on 2020-06-20 | In MYSQL

sudo mkdir -p /data/aiaas/config/mysql/
sudo bash -c ‘cat >> /data/aiaas/config/mysql/my.cnf << EOF
[mysqld]
#

Remove leading # and set to the amount of RAM for the most important data

cache in MySQL. Start at 70% of total RAM for dedicated server, else 10%.

innodb_buffer_pool_size = 128M

#

Remove leading # to turn on a very important data integrity option: logging

changes to the binary log between backups.

log_bin

#

Remove leading # to set options mainly useful for reporting servers.

The server defaults are faster for transactions and fast SELECTs.

Adjust sizes as needed, experiment to find the optimal values.

join_buffer_size = 128M

sort_buffer_size = 2M

read_rnd_buffer_size = 2M

skip-host-cache
skip-name-resolve
datadir=/var/lib/mysql
socket=/var/lib/mysql/mysql.sock
secure-file-priv=/var/lib/mysql-files
user=mysql
max_allowed_packet=100M

Disabling symbolic-links is recommended to prevent assorted security risks

symbolic-links=0

log-error=/var/log/mysqld.log
pid-file=/var/run/mysqld/mysqld.pid
sql_mode=ONLY_FULL_GROUP_BY,STRICT_TRANS_TABLES,ERROR_FOR_DIVISION_BY_ZERO,NO_AUTO_CREATE_USER,NO_ENGINE_SUBSTITUTION
EOF’

sudo bash -c ‘cat >> /data/aiaas/shells/docker-compose-mysql.yml << EOF
version: ‘3’

services:

mysql:
image: hub.lingban.cn/mysql:5.7
restart: always
privileged: true
volumes:

   - /data/aiaas/data/mysql/mysql_init:/docker-entrypoint-initdb.d
   - /data/aiaas/data/mysql/data:/var/lib/mysql
   - /data/aiaas/config/mysql/my.cnf:/etc/my.cnf
   - /data/aiaas/logs/mysql/log:/var/log
environment:
   TZ: Asia/Shanghai
   MYSQL_ROOT_PASSWORD: Lingban2018_AIAAS
ports:
  - "3306:3306"
logging:
  options:
    max-size: '100m'
    max-file: '10'

volumes:
app_net:
external: true
EOF’

cd /data/aiaas/shells/
sudo docker-compose -f docker-compose-mysql.yml up -d

# mysql # DB
